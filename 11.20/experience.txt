题目要求使用冒泡排序 而我使用了选择排序 二者的区别为
冒泡排序
for(i=0;i<n-1;i++){
    for(j=0;j<n-i-1;j++){//减i是因为最大值已经沉在后面i个了
        if(number[j]>number[j+1]){
            t=number[j];
            number[j]=number[j+1];
            number[j+1]=t;
        }
    }
}

选择排序
for(i=0;i<n-1;i++){
    for(j=i+1;j<n;j++){//每次从第一个开始 比较这个数与这个数之后所有数字的大小
        if(number[i]>number[j]){
            t=number[i];
            number[i]=number[j];
            number[j]=t;
        }
    }
}
